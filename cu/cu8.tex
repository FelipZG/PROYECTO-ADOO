\begin{UseCase}{CU8}{Agregar un Nuevo Empleado.}{
		Se requiere una manera sencilla e intuitiva para que el dueño pueda agregar a sus empleados.
	}
		\UCitem{Versión}{\color{Gray}0.1}
		\UCitem{Autor}{\color{Gray}Vázquez Cruz Fernando Darwin }
		\UCitem{Supervisa}{\color{Gray}}
		\UCitem{Actor}{\hyperlink{Alumno}{Dueño}}
		\UCitem{Propósito}{Agregar un empleado que laborará en la empresa.}
		\UCitem{Entradas}{Datos del empleado.}
		\UCitem{Origen}{Teclado, Mouse}
		\UCitem{Salidas}{Mensaje de confirmación de registro.}
		\UCitem{Destino}{Pantalla}
		\UCitem{Precondiciones}{}
		\UCitem{Postcondiciones}{Habrá un nuevo  empleado registrado en el sistema.}
		\UCitem{Errores}{El empleado no puede ser agregado.}
		\UCitem{Tipo}{Caso de uso primario.}
		\UCitem{Observaciones}{}
		\UCitem{Estado}{Revisión}
	\end{UseCase}
%--------------------------------------
	\begin{UCtrayectoria}{Principal}
		\UCpaso Se incluye el caso de uso \UCref{CU1}.
		\UCpaso Despliega la \IUref {IU1} {Pantalla Principal}.
		\UCpaso[\UCactor] Ve los empleados disponibles dando clic en el\Ibutton {+ Empleados }.
		\UCpaso Despliega los tipos de empleado existentes en el sistema abajo del \Ibutton {+ Empleados }.
		\UCpaso[\UCactor] Da clic en el \Ibutton {+ Cajero } o en el \Ibutton {+ Supervisor } según el tipo de empleado a agregar.
		\UCpaso Despliega la \IUref {IU18} {Tabla de Cajeros} o la \IUref {IU13} {Tabla de Supervisores} según el tipo de empleado a agregar.
		\UCpaso[\UCactor] Independientemente del tipo de empleado a agregar presiona el \Ibutton {+ Nuevo }.
		\UCpaso Despliega la \IUref {IU7} {Formulario Cajero} o la \IUref {IU2} {Formulario Supervisor} según el tipo de empleado a agregar.
		\UCpaso [\UCactor] Llena los campos requeridos para registrar un nuevo empleado. 
		\UCpaso[\UCactor] Da clic en el \Ibutton {+ Guardar } .
		\UCpaso Despliega el mensaje \bf {+ MSG0}. \Trayref{A}
		\UCpaso Guarda los datos del nuevo empleado y actualiza la lista de empleados.
		\UCpaso[\UCactor] Da clic en el \Ibutton {+ Ok }.
		\UCpaso Re-direcciona al \UCactor a la la \IUref {IU18} {Tabla de Cajeros} o la \IUref {IU13} {Tabla de Supervisores} según el tipo de empleado agregado.
	\end{UCtrayectoria}


		\begin{UCtrayectoriaA}{A}{El Empleado no puede ser agregado}
			\UCpaso Despliega el mensaje \bf {+ MSG1 }.
			\UCpaso[\UCactor] Da clic en el \Ibutton {+ Ok }.
			\UCpaso Continua en el paso 7 del \UCref{CU8}.
		\end{UCtrayectoriaA}
