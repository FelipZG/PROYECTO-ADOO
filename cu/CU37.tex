\begin{UseCase}{CU37}{Aplicar un descuento}{
		El sistema detecta descuentos aplicabres a los medicamentos.
	}
		\UCitem{Versión}{\color{Gray}0.1}
		\UCitem{Autor}{\color{Gray}Felipe Zamora Gachuz}
		\UCitem{Supervisa}{\color{Gray}.}
		\UCitem{Actor}{Cajero}
		\UCitem{Propósito}{El sistema permite un descuento valido}%
		\UCitem{Entradas}{Una venta valida}%
		\UCitem{Origen}{Sistema}
		\UCitem{Salidas}{ }
		\UCitem{Destino}{ }
		\UCitem{Precondiciones}{Tener la venta valida, estar en las pantallas del cajero}
		\UCitem{Postcondiciones}{Se genera nuevos datos del precios para la venta}
		\UCitem{Errores}{Error de conexion, los descuentos no son validos para esos productos}
		\UCitem{Observaciones}{}
		\UCitem{Estado}{En revision}
