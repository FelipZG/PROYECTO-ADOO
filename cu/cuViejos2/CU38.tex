	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Tiene que cumplir el caso de uso \IUref{IU1}{Login} en el apartado del cajero.
		\UCpaso[\UCactor] Tiene que estar en la pestaña de nuevo Proveedor.
		\UCpaso El sistema despliega un formulario de registro de nuevo proveedor\IUref{IU8}{ProveedorFormulario}. 
		\UCpaso [\UCactor] Proporciona los datos requeridos en el formulario \IUref{IU8}{ProveedorFormulario}.
		\UCpaso Verifica los datos ingresados por el cajero \Trayref{A}.
		\UCpaso Presiona el boton Aceptar.
	\end{UCtrayectoria}
