		\UCitem{Salidas}{-}

	\begin{UCtrayectoria}{Principal}
		\UCpaso La trayectoria se Extiende del caso de uso 'Hacer una devolución al cliente' \UCref{CU13}

		\UCpaso [\UCactor] Selecciona el boton de \IUbutton{Aceptar} \Trayref{A}
